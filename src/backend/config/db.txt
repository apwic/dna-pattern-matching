CREATE TABLE Penyakit (
	IdPenyakit INT AUTO_INCREMENT,
	NamaPenyakit VARCHAR(255) UNIQUE NOT NULL,
	Sekuens VARCHAR(60000) NOT NULL,
	PRIMARY KEY (IDPenyakit)
);

CREATE TABLE TesDNA (
	IDPengguna INT AUTO_INCREMENT,
	NamaPengguna VARCHAR(255) NOT NULL,
	Penyakit VARCHAR(255) NOT NULL,
	Status BOOLEAN NOT NULL,
	Kemiripan INT CHECK(Kemiripan BETWEEN 0 AND 100),
	Tanggal DATETIME NOT NULL,
	Sekuens VARCHAR(60000) NOT NULL,
	PRIMARY KEY (IDPengguna),
	FOREIGN KEY (Penyakit) REFERENCES Penyakit (NamaPenyakit) ON DELETE RESTRICT ON UPDATE CASCADE
);
